CREATE TABLE PRODUCT (
    Product_ID NUMBER PRIMARY KEY,
    Product_Name VARCHAR2(50) NOT NULL,
    Category VARCHAR2(30) NOT NULL,
    Price NUMBER(10,2) CHECK (Price > 0),
    Stock NUMBER CHECK (Stock >= 0)
);

CREATE TABLE CUSTOMER (
    Customer_ID NUMBER PRIMARY KEY,
    Customer_Name VARCHAR2(50) NOT NULL,
    City VARCHAR2(30) NOT NULL,
    Mobile_No VARCHAR2(15) UNIQUE
);

CREATE TABLE ORDERS (
    Order_ID NUMBER PRIMARY KEY,
    Product_ID NUMBER,
    Customer_ID NUMBER,
    Quantity NUMBER CHECK (Quantity > 0),
CONSTRAINT fk_product
        FOREIGN KEY (Product_ID)
        REFERENCES PRODUCT(Product_ID),
CONSTRAINT fk_customer
        FOREIGN KEY (Customer_ID)
        REFERENCES CUSTOMER(Customer_ID)
);

SELECT UPPER(Customer_Name) AS Customer_Name
FROM CUSTOMER;

SELECT LOWER(Product_Name) AS Product_Name
FROM PRODUCT;

SELECT Product_Name, LENGTH(Product_Name) AS Name_Length
FROM PRODUCT;

SELECT CONCAT(Customer_Name, City) AS Customer_City
FROM CUSTOMER;

SELECT O.Order_ID,
       P.Product_Name,
       O.Quantity,
       P.Price,
       (O.Quantity * P.Price) AS Total_Cost
FROM ORDERS O
JOIN PRODUCT P
ON O.Product_ID = P.Product_ID;

SELECT *
FROM PRODUCT
ORDER BY Price DESC;
