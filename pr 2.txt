CREATE TABLE BOOK (
    Book_ID NUMBER(5) PRIMARY KEY,
    Book_Name VARCHAR2(100) NOT NULL,
    Author VARCHAR2(50) NOT NULL,
    Price NUMBER(8,2) CHECK (Price > 0),
    Category VARCHAR2(30)
);
desc book;

CREATE TABLE MEMBER (
    Member_ID NUMBER(5) PRIMARY KEY,
    Member_Name VARCHAR2(50) NOT NULL,
    Phone VARCHAR2(10) UNIQUE,
    City VARCHAR2(30)
);
desc member;
CREATE TABLE ISSUE (
    Issue_ID NUMBER(5) PRIMARY KEY,
    Book_ID NUMBER(5),
    Member_ID NUMBER(5),
    Issue_Date DATE
);
desc issue;
